{ return (widget_state & f) != 0; }

{ return (widget_flags & f) != 0; }

{ return widget_flags; }

{ widget_flags |= f; }

{ widget_flags &= ~f; }

inline void QWidget::constPolish() const
{
    if ( !testWState(WState_Polished) ) {
        QWidget* that = (QWidget*) this;
        that->polish();
        that->setWState(WState_Polished); // be on the safe side...
    }
}
